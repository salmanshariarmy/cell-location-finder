async function findLocation(){ let response = await fetch("cells.csv"); let text = await response.text(); let rows = text.trim().split("\n"); let mcc = document.getElementById("mcc").value; let mnc = document.getElementById("mnc").value; let lac = document.getElementById("lac").value; let cid = document.getElementById("cid").value; for(let i = 1; i < rows.length; i++){ let data = rows[i].split(","); if( data[0] == mcc && data[1] == mnc && data[2] == lac && data[3] == cid ){ let lat = data[4]; let lon = data[5]; map.setView([lat,lon],15); L.marker([lat,lon]) .addTo(map) .bindPopup( "Cell Location
"+ lat+","+lon ) .openPopup(); document.getElementById("result").innerHTML = "Latitude: "+lat+ "
Longitude: "+lon+ "
Open Google Maps"; return; } } document.getElementById("result").innerHTML = "Cell not found"; }
